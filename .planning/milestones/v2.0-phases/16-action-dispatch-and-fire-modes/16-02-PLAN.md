```python
class FireMode(Enum):
    TAP = "tap"
    HOLD_KEY = "hold_key"

@dataclass(frozen=True)
class Action:
    key_string: str
    fire_mode: FireMode
    gesture_name: str
    modifiers: list      # list[Key]
    key: object          # Key | str

class ActionResolver:
    def __init__(self, right_actions: dict[str, Action], left_actions: dict[str, Action],
                 right_compound: dict[tuple[str, str], Action], left_compound: dict[tuple[str, str], Action]): ...
    def set_hand(self, handedness: str) -> None: ...
    def resolve(self, gesture_name: str) -> Optional[Action]: ...
    def resolve_compound(self, gesture_name: str, direction: str) -> Optional[Action]: ...

class ActionDispatcher:
    def __init__(self, sender: KeystrokeSender, resolver: ActionResolver): ...
    def dispatch(self, signal: OrchestratorSignal) -> None: ...
    def release_all(self) -> None: ...
```
