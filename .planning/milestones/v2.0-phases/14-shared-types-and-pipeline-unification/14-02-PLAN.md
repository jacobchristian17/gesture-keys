```python
def run_preview_mode(args):
    config = load_config(args.config)
    print_banner(config, args.config)

    # Setup logging
    log_level = logging.DEBUG if args.debug else logging.INFO
    logging.basicConfig(
        format="[%(asctime)s] %(message)s",
        datefmt="%H:%M:%S",
        level=log_level,
    )

    pipeline = Pipeline(args.config)
    pipeline.start()

    prev_time = time.perf_counter()
    fps = 0.0
    try:
        while True:
            # FPS calculation
            current_time = time.perf_counter()
            dt = current_time - prev_time
            if dt > 0:
                fps = 1.0 / dt
            prev_time = current_time

            result = pipeline.process_frame()
            if not result.frame_valid:
                continue

            # Per-frame debug logging (preview-only)
            if args.debug and result.landmarks:
                logger.debug(
                    "FRAME raw=%s smooth=%s state=%s",
                    result.raw_gesture.value if result.raw_gesture else "None",
                    result.gesture.value if result.gesture else "None",
                    result.debounce_state.value,
                )

            # Preview rendering
            if args.preview:
                frame = pipeline.last_frame
                if result.landmarks:
                    draw_hand_landmarks(frame, result.landmarks)
                gesture_label = result.gesture.value if result.gesture else None
                render_preview(
                    frame, gesture_label, fps,
                    debounce_state=result.debounce_state.value,
                    handedness=result.handedness,
                )

                key = cv2.waitKey(1) & 0xFF
                if key == 27:
                    break
                try:
                    if cv2.getWindowProperty("Gesture Keys", cv2.WND_PROP_VISIBLE) < 1:
                        break
                except cv2.error:
                    break

    except KeyboardInterrupt:
        pass
    finally:
        pipeline.stop()
        if args.preview:
            cv2.destroyAllWindows()
```

```python
def _detection_loop(self) -> None:
    while not self._shutdown.is_set():
        if not self._active.wait(timeout=0.5):
            continue
        if self._shutdown.is_set():
            break

        try:
            config = load_config(self._config_path)
        except Exception as e:
            logger.error("Failed to load config: %s", e)
            time.sleep(1.0)
            continue

        pipeline = Pipeline(self._config_path)
        pipeline.start()
        try:
            while self._active.is_set() and not self._shutdown.is_set():
                pipeline.process_frame()
        finally:
            pipeline.stop()
```

This preserves the existing behavior: tray mode tears down and recreates the full pipeline on each active/inactive cycle. Pipeline.start()/stop() maps to the current create/teardown pattern.

Update imports: remove all individual component imports. Add `from gesture_keys.pipeline import Pipeline`. Keep imports for: `load_config` (for the error-handling try/except before Pipeline creation), `logging`, `os`, `threading`, `time`, `pystray`, `PIL`.

Note on the config load before Pipeline: The existing tray code loads config and handles the error case (sleep 1s, continue) before creating components. We keep this pattern because Pipeline.__init__ also loads config but doesn't have the try/sleep/continue error handling. The double-load is cheap and preserves the error resilience pattern. Alternatively, Pipeline.__init__ could accept the pre-loaded config, but the research recommends accepting path string for simplicity.
